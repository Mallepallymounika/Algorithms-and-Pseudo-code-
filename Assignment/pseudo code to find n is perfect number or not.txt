//pseudo code to find n is perfect number or not


begin
    
      declare the variables sum,n,i
      read the the positive integer
      initialize sum=0
  repeat the steps 1,2 
     while (i<n)
     1) if (n%i==0)
      {
       sum=sum+i
        }
      2) i=i+1
       
     
if (sum == n)
{
   print "n is perfect number"
else 
   print "n is not a perfect number"
}

 stop
